					"height", //animated property name
					0.0, //begin
					100.0, //end
					3000, //duratuin
					duit_animations.OnEnter, //trigger
					duit_animations.Forward, //method
					false, //repeat
					"", //linear curve by default
				),
			},
		},
		"", //id

	)
